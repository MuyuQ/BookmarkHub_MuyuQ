            }

            if (b.url && !localUrls.has(b.url)) {

            } else if (!b.url && b.children) {
                findBookmarksToCreate(b.children, b.title);
            }
        }

    findBookmarksToCreate(mergedBookmarks);
    
    if (bookmarksToCreate.length === 0) {

    let browserType = BrowserType.CHROME;
    if (localBookmarks[0]?.id === ROOT_NODE_IDS.ROOT[1]) {
        browserType = BrowserType.FIREFOX;
    }
    
    const folderIdMap = new Map<string, string>();
    
    function collectFolderIds(bookmarks: BookmarkInfo[]): void {
        for (const b of bookmarks) {
            if (!b.url && b.id && b.title) {
                folderIdMap.set(b.title, b.id);
            }
            if (b.children) {
                collectFolderIds(b.children);
            }
        }
    }
    collectFolderIds(localBookmarks);
    
    let createdCount = 0;
    const defaultParentId = browserType === BrowserType.FIREFOX 
        ? ROOT_NODE_IDS.UNFILED[1] 
        : ROOT_NODE_IDS.UNFILED[0];
    
    for (const { bookmark, parentTitle } of bookmarksToCreate) {
        try {
            let parentId: string = defaultParentId;
            
            if (parentTitle) {

                if (foundId) {

            await browser.bookmarks.create({

                url: bookmark.url, 
                error: err 
            });
        }
    }
    
    logger.info(`syncLocalBookmarks: Created ${createdCount} bookmarks`);
    return createdCount;
}

 * @param result - 同步结果
 */
async function saveSyncStatus(result: SyncResult): Promise<void> {
    await browser.storage.local.set({
        [STORAGE_KEYS.LAST_SYNC_TIME]: result.timestamp,
        lastSyncDirection: result.direction,
        [STORAGE_KEYS.LAST_SYNC_STATUS]: result.status,
        [STORAGE_KEYS.LAST_SYNC_ERROR]: result.errorMessage || '',
        [STORAGE_KEYS.LOCAL_COUNT]: result.localCount,
        [STORAGE_KEYS.REMOTE_COUNT]: result.remoteCount
    });
}
 
/**
 * 发送冲突通知
 * 当检测到冲突时，发送系统通知
 * 
 * @param conflicts - 冲突列表
 */
export async function notifyConflict(conflicts: ConflictInfo[]): Promise<void> {
    // 没有冲突则跳过
    if (conflicts.length === 0) return;
    
    // 获取设置，检查是否启用通知
    const setting = await Setting.build();
    if (!setting.enableNotify) return;
    
    // 发送系统通知
    await browser.notifications.create({
        type: 'basic',
        iconUrl: '../assets/icon.png',
        title: 'Sync Conflict',
        message: `${conflicts.length} conflicts detected`
    });
}
